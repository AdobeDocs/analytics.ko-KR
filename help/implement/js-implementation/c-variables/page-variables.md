* [browserWidth]
* [campaign]
* [channel]
* [colorDepth]
* [connectionType]
* [cookiesEnabled]
* [dc]

* [javaEnabled]
* [javascriptVersion]
* [linkName]
* [linkType]
* [List Props]

* [maxDelay]
* [mediaLength]
* [mediaName]
* [mediaPlayer]
* [mediaSession]
* [Media.trackEvents]

* [resolution]
